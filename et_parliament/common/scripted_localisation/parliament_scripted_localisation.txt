defined_text = {
	name = GetParliamentNumOfYesVotes
	
	text = {
		trigger = {
			set_temp_variable = { parliament_num_of_yes_votes_temp = 0 }
			all_of = {
				array = parliament_seats_yes_no_array
				value = v
				if = {
					limit = {
						check_variable = { v = 1 }
					}
					add_to_temp_variable = { parliament_num_of_yes_votes_temp = 1 }
				}
			}
			always = yes
		}
		localization_key = string_parliament_num_of_yes_votes
	}
}

defined_text = {
	name = GetParliamentNumOfNoVotes

		trigger = {

				}
