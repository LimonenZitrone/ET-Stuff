# Voting Text
defined_text = {
	name = GetCurrentVotingText
	
	text = {
		trigger = {
			has_global_flag = un_council_vote_active
		}
		localization_key = string_un_current_vote_text
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = string_un_no_current_vote_text
	}
}

	name = GetCurrentVoteIndicatorTooltip1
	
	text = {
		trigger = {
			NOT = { check_variable = { global.un_council_members^0 = 0 } }

			var:selected_un_council_member = {
				has_country_flag = un_council_vote_yes

# Opinion Indicator Tooltip:
defined_text = {
	name = GetOpinionIndicatorTooltip1
	
	text = {
		trigger = {
			NOT = { check_variable = { global.un_council_members^0 = 0 } }
			set_temp_variable = { selected_un_council_member = global.un_council_members^0 }
			var:selected_un_council_member = {
				tag = ROOT
			}
		}
		localization_key = string_opinion_indicator_thats_us
	}
	text = {
		trigger = {
			NOT = { check_variable = { global.un_council_members^0 = 0 } }
			set_temp_variable = { selected_un_council_member = global.un_council_members^0 }
			var:selected_un_council_member = {
				has_opinion = {
					target = ROOT
					value > 0
				}
			}
			var:selected_un_council_member = {
				set_temp_variable = { selected_un_council_member_opinion = opinion@ROOT }
			}
		}
		localization_key = string_opinion_indicator_positive
	}
	text = {
		trigger = {
			NOT = { check_variable = { global.un_council_members^0 = 0 } }
			set_temp_variable = { selected_un_council_member = global.un_council_members^0 }
			var:selected_un_council_member = {
				set_temp_variable = { selected_un_council_member_opinion = opinion@ROOT }
			}
		}
		localization_key = string_opinion_indicator_negative
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = string_indicator_invalid
	}
}
