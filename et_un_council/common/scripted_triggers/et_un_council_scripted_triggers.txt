set_color_indicators_for_un_council_member_1 = {
	if = {
		limit = {
			NOT = { check_variable = { global.un_council_members^0 = 0 } }
		}
		var:global.un_council_members^0 = {
			if = {
				limit = {
					has_country_flag = un_council_vote_yes
				}
				set_temp_variable = { current_vote_indicator_1_frame = 1 }
			}
			else_if = {
				limit = {
					has_country_flag = un_council_vote_no
				}
				set_temp_variable = { current_vote_indicator_1_frame = 2 }
			}
			else = {
				set_temp_variable = { current_vote_indicator_1_frame = 4 }
			}
			if = {
				limit = {
					OR = {
						tag = ROOT
						has_opinion = {
							target = ROOT
							value > 0
						}
					}
				}
				set_temp_variable = { opinion_indicator_1_frame = 1 }
			}
			else = {
				set_temp_variable = { opinion_indicator_1_frame = 2 }
			}
			if = {
				limit = {
					is_in_faction = yes
					is_in_faction_with = ROOT
				}
				set_temp_variable = { faction_indicator_1_frame = 3 }
			}
			else_if = {
				limit = {
					is_in_faction = yes
				}
				set_temp_variable = { faction_indicator_1_frame = 2 }
			}
			else = {
				set_temp_variable = { faction_indicator_1_frame = 4 }
			}
		}
	}
	else = {
		set_temp_variable = { current_vote_indicator_1_frame = 4 }
		set_temp_variable = { opinion_indicator_1_frame = 4 }
		set_temp_variable = { faction_indicator_1_frame = 4 }
	}
}
